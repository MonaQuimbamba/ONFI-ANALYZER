176 bool ONFIAnalyzer::AddFrame(FrameType frame_type,
177  U64 start,
178  U64 end,
179  U64 data1,
180  U64 data2,
181  U8 flags)
182 {
183  // NOTE: end - start must be > 0 or Logic crashes when trying to zoom to the
184  // frame
185  if (start >= end) {
186  return false;
187  }
188  Frame frame{};
189  frame.mStartingSampleInclusive = start;
190  frame.mEndingSampleInclusive = end;
191  frame.mType = frame_type;
192  frame.mData1 = data1;
193  frame.mData2 = data2;
194  frame.mFlags = flags;
195  mResults->AddFrame(frame);
196  ReportProgress(frame.mEndingSampleInclusive);
197  return true;
